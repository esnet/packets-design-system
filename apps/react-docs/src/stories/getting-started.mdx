    Install the packages:

```bash
npm install @esnet/packets-ui-css @esnet/packets-ui-react
```

```css
@import "@esnet/packets-ui-css/dist/styles.css";
```

    # Usage

    Import components and use them in your React application:

```tsx
import { PktsButton, PktsInputText, PktsAlert } from '@esnet/packets-ui-react';

function MyComponent() {
  const [value, setValue] = useState('');

  return (
    <div className="packets">
      <PktsButton variant="primary" onClick={() => alert('Clicked!')}>
        Click Me
      </PktsButton>

      <PktsInputText
        placeholder="Enter text..."
        value={value}
        onChange={(e) => setValue(e.target.value)}
      />

      <PktsAlert variant="info" title="Info Alert">
        This is an informational message.
      </PktsAlert>
    </div>
  );
}
```

    # Theme

    Wrap your application with the `packets` class to enable Packets styling:

```tsx
function App() {
  return (
    <div className="packets">
      {/* Your application components */}
    </div>
  );
}
```

    For dark mode, add the `dark` class:

```tsx
function App() {
  const [darkMode, setDarkMode] = useState(false);

  return (
    <div className={`packets ${darkMode ? 'dark' : 'light'}`}>
      {/* Your application components */}
    </div>
  );
}
```

    # TypeScript Support

    All React components are written in TypeScript and provide full type definitions. You'll get complete IntelliSense and type checking:

```tsx
import { PktsButton } from '@esnet/packets-ui-react';
import type { PktsButtonProps } from '@esnet/packets-ui-react';

const MyButton: React.FC = () => {
  return (
    <PktsButton
      variant="primary"  // TypeScript autocompletes: "primary" | "secondary" | "tertiary"
      size="medium"      // TypeScript autocompletes: "small" | "medium" | "large"
      onClick={(e) => {
        console.log(e.currentTarget);
      }}
    >
      Click Me
    </PktsButton>
  );
};
```
