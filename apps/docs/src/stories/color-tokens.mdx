import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs/blocks";
import * as tokens from "@esnet/esnet-tokens/tokens.json";

<Meta title="Design Tokens/Colors" />

#### Design Tokens
# Colors
<br />

Here are the inital set of color ramps and color alias tokens

<ColorPalette>
  {
    Object.keys(tokens.color).map(colorGroup => {
      const currentColorGroup = tokens.color[colorGroup];
      const isCore = colorGroup === "core";

      if(isCore) {
        return Object.keys(currentColorGroup).map(colorRamp => {
          const currentColorRamp = currentColorGroup[colorRamp];
          const formattedColorGroups = {};

          Object.keys(currentColorRamp).forEach(color => {
            formattedColorGroups[`--esnet-color-core-${colorRamp}-${color}`] = currentColorRamp[color].value
          })

          return (<ColorItem
                      title={`color.${colorGroup}.${colorRamp}`}
                      subtitle={`The ${colorRamp} color ramp.`}
                      colors={formattedColorGroups}
                    />)
        })
      } else {
          const formattedColorGroups = {};

          Object.keys(currentColorGroup).forEach(color => {
            formattedColorGroups[`--esnet-color-${colorGroup}-${color}`] = currentColorGroup[color].value
          })

          return (<ColorItem
                      title={`color.${colorGroup}`}
                      subtitle={`The ${colorGroup} theme alias tokens.`}
                      colors={formattedColorGroups}
                    />)
      }


    })

}

</ColorPalette>
