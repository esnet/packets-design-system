    # Usage

    Import components and use them in your React application:

```tsx
import { ESButton, ESInputText, ESAlert } from '@esnet/packets-ui-react';

function MyComponent() {
  const [value, setValue] = useState('');

  return (
    <div className="packets">
      <ESButton variant="primary" onClick={() => alert('Clicked!')}>
        Click Me
      </ESButton>

      <ESInputText
        placeholder="Enter text..."
        value={value}
        onChange={(e) => setValue(e.target.value)}
      />

      <ESAlert variant="info" title="Info Alert">
        This is an informational message.
      </ESAlert>
    </div>
  );
}
```

    # TypeScript Support

    All React components are written in TypeScript and provide full type definitions. You'll get complete IntelliSense and type checking:

```tsx
import { ESButton } from '@esnet/packets-ui-react';
import type { ESButtonProps } from '@esnet/packets-ui-react';

// Fully typed component props
const MyButton: React.FC = () => {
  return (
    <ESButton
      variant="primary"  // TypeScript will autocomplete: "primary" | "secondary" | "tertiary"
      size="medium"      // TypeScript will autocomplete: "small" | "medium" | "large"
      onClick={(e) => {
        // e is fully typed as React.MouseEvent
        console.log(e.currentTarget);
      }}
    >
      Click Me
    </ESButton>
  );
};
```

    # Theme Setup

    Wrap your application with the `packets` class to enable Packets styling:

```tsx
function App() {
  return (
    <div className="packets">
      {/* Your application components */}
    </div>
  );
}
```

    For dark mode, add the `dark` class:

```tsx
function App() {
  const [darkMode, setDarkMode] = useState(false);

  return (
    <div className={`packets ${darkMode ? 'dark' : 'light'}`}>
      {/* Your application components */}
    </div>
  );
}
```
